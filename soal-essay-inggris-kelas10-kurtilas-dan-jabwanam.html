Nomor 1b

[tex]x_1<x_2=(x_1+2x_2)^2-2x_2[/tex]

pandang

[tex]x_2=(x_1+2x_2)^2-2x_2=(x_1^2+4x_1x_2+4x_2^2)-2x_2[/tex]

[tex]0=(x_1^2+4x_1x_2+4x_2^2)-3x_2[/tex]

[tex]0=4[x_2^2]+(4x_1-3)[x_2]+x_1^2[/tex]

dengan rumus abc, dapat dicari solusi [tex]x_2[/tex]

[tex]x_2=\frac{-(4x_1-3) \pm\sqrt{(4x_1-3)^2-4(4)(x_1^2)} }{2(4)}[/tex]

[tex]=\frac{1}{8} (-4x_1+3 \pm\sqrt{16x_1^2-24x_1+9-16x_1^2} )[/tex]

[tex]=\frac{1}{8} (-4x_1+3 \pm \sqrt{9-24x_1} )[/tex]

[tex]=-\frac{1}{2} x_1+\frac{3}{8} \pm\sqrt{9-24x_1}[/tex]          dimana [tex]x_1 \leq \frac{3}{8}[/tex]

sehingga

[tex]x_1<-\frac{1}{2} x_1+\frac{3}{8} \pm\sqrt{9-24x_1}[/tex]

[tex]\frac{3}{2}x_1-\frac{3}{8} < \pm\sqrt{9-24x_1}[/tex]

karena

[tex]x_1\leq \frac{3}{8} \\\frac{3}{2} x_1-\frac{3}{8} \leq \frac{3}{2} \cdot \frac{3}{8} -\frac{3}{8}=\frac{3}{16}[/tex]

bagi dua kasus

kasus 1:

untuk     [tex]0\leq \frac{3}{2} x_1-\frac{3}{8} \leq \frac{3}{16}[/tex]

untuk     [tex]\frac{3}{2} x_1-\frac{3}{8} <0[/tex]
